-- Patient Analytics Optimization
WITH MonthlyTrend AS (
  SELECT
    DATENAME(MONTH, VisitDate)
      AS MonthName,
    COUNT(*) AS Visits,
    COUNT(DISTINCT PatientID)
      AS UniquePatients
  FROM Appointments
  WHERE VisitDate >=
    DATEADD(MONTH,-12,GETDATE())
  GROUP BY DATENAME(MONTH,VisitDate)
)
SELECT * FROM MonthlyTrend
ORDER BY Visits DESC;
