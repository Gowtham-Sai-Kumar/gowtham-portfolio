Query 1 — Patient Count by Clinic
Uses JOIN + GROUP BY + ORDER BY · Execution: ~1.2s
Optimized
-- Get total patients per clinic with conversion rate SELECT c.ClinicName, COUNT(p.PatientID) AS TotalPatients, SUM(CASE WHEN a.Status = 'Completed' THEN 1 ELSE 0 END) AS CompletedConsults, ROUND( SUM(CASE WHEN a.Status = 'Completed' THEN 1.0 ELSE 0 END) / COUNT(p.PatientID) * 100, 2 ) AS ConversionRate FROM Patients p INNER JOIN Clinics c ON p.ClinicID = c.ClinicID LEFT JOIN Appointments a ON p.PatientID = a.PatientID GROUP BY c.ClinicName ORDER BY TotalPatients DESC;
Query 2 — Monthly Visit Trend
Uses DATEPART + YEAR function + Indexed date columns
Aggregation
-- Monthly patient visit trend for the past 12 months SELECT YEAR(VisitDate) AS VisitYear, MONTH(VisitDate) AS VisitMonth, DATENAME(MONTH, VisitDate) AS MonthName, COUNT(*) AS TotalVisits, COUNT(DISTINCT PatientID) AS UniquePatients FROM Appointments WHERE VisitDate >= DATEADD(MONTH, -12, GETDATE()) GROUP BY YEAR(VisitDate), MONTH(VisitDate), DATENAME(MONTH, VisitDate) ORDER BY VisitYear, VisitMonth;
Stored Procedure — VC Session Summary
Parameterized stored procedure with error handling · Production-ready
Stored Proc
CREATE PROCEDURE usp_GetVCSummary @FromDate DATE = NULL, @ToDate DATE = NULL, @ClinicID INT = NULL AS BEGIN SET NOCOUNT ON; -- Default to current month if no dates given SET @FromDate = ISNULL(@FromDate, DATEADD(DAY, 1, EOMONTH(GETDATE(), -1))); SET @ToDate = ISNULL(@ToDate, GETDATE()); SELECT c.ClinicName, COUNT(*) AS TotalSessions, SUM(CASE WHEN v.VCStatus = 'Completed' THEN 1 ELSE 0 END) AS Completed, SUM(CASE WHEN v.VCStatus = 'PatientMissed'THEN 1 ELSE 0 END) AS PatientMissed, SUM(CASE WHEN v.VCStatus = 'DoctorOnly' THEN 1 ELSE 0 END) AS DoctorOnlyJoined FROM VCSessions v INNER JOIN Clinics c ON v.ClinicID = c.ClinicID WHERE v.SessionDate BETWEEN @FromDate AND @ToDate AND (@ClinicID IS NULL OR v.ClinicID = @ClinicID) GROUP BY c.ClinicName ORDER BY Completed DESC; END;
Query 3 — Missed VC with Edge Cases
Handles NULL mobile, duplicate entries, failed sessions
Edge Cases
-- Find missed VCs with patient details, handle NULLs and duplicates WITH MissedSessions AS ( SELECT p.PatientID, p.PatientName, ISNULL(p.Mobile, 'N/A') AS Mobile, -- Handle missing mobile c.ClinicName, v.SessionDate, v.VCStatus, ROW_NUMBER() OVER ( PARTITION BY p.PatientID, v.SessionDate ORDER BY v.SessionID DESC ) AS rn -- Remove duplicates FROM VCSessions v INNER JOIN Patients p ON v.PatientID = p.PatientID INNER JOIN Clinics c ON v.ClinicID = c.ClinicID WHERE v.VCStatus IN ('PatientMissed', 'Failed') ) SELECT * FROM MissedSessions WHERE rn = 1 -- Keep latest record only ORDER BY SessionDate DESC;
